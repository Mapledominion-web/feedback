    <script>
        // ================================================================
        // Global: store feedback data until broker choice is made
        // ================================================================
        let storedFeedback = {
            fullName: '',
            trialExperience: '',
            satisfaction: '',
            willingness: '',
            tradingGoal: '',
            valuedTraits: []
        };

        /**
         * Update the step indicator
         */
        function updateSteps(activeStep) {
            const steps = document.querySelectorAll('#stepIndicator .step');
            steps.forEach((step, index) => {
                const num = index + 1;
                step.classList.remove('active', 'completed');
                if (num < activeStep) step.classList.add('completed');
                else if (num === activeStep) step.classList.add('active');
            });
        }

        /**
         * Store feedback and show broker list (Step 2)
         */
        function storeFeedbackAndShowBrokers() {
            const fullName = document.getElementById('fullName');
            const trialExp = document.getElementById('trialExperience');
            const satisfaction = document.getElementById('satisfaction');
            const willingness = document.getElementById('willingness');
            const goal = document.getElementById('goal');
            let valid = true;

            // Validation
            if (!fullName.value.trim()) { valid = false; fullName.style.borderColor = 'var(--theme-color-three)'; } 
            else { fullName.style.borderColor = ''; }

            if (!trialExp.value.trim()) { valid = false; trialExp.style.borderColor = 'var(--theme-color-three)'; } 
            else { trialExp.style.borderColor = ''; }

            if (!satisfaction.value) { valid = false; satisfaction.style.borderColor = 'var(--theme-color-three)'; } 
            else { satisfaction.style.borderColor = ''; }

            if (!willingness.value) { valid = false; willingness.style.borderColor = 'var(--theme-color-three)'; } 
            else { willingness.style.borderColor = ''; }

            if (!goal.value.trim()) { valid = false; goal.style.borderColor = 'var(--theme-color-three)'; } 
            else { goal.style.borderColor = ''; }

            if (!valid) {
                alert('Please fill in all required fields (marked with *).');
                return;
            }

            // Collect traits
            const selectedTraits = [];
            document.querySelectorAll('#traitsGroup input[type="checkbox"]:checked').forEach(cb => {
                selectedTraits.push(cb.value);
            });

            // Store all data
            storedFeedback.fullName = fullName.value.trim();
            storedFeedback.trialExperience = trialExp.value.trim();
            storedFeedback.satisfaction = satisfaction.value;
            storedFeedback.willingness = willingness.value;
            storedFeedback.tradingGoal = goal.value.trim();
            storedFeedback.valuedTraits = selectedTraits;

            // Hide feedback, show brokers
            document.getElementById('feedback').style.display = 'none';
            document.getElementById('brokerSection').style.display = 'block';
            updateSteps(2);
            document.getElementById('brokerSection').scrollIntoView({ behavior: 'smooth', block: 'start' });
        }

        /**
         * Handle broker choice: send all data to Formspree, then show confirmation (Step 3)
         */
        function selectBroker(choice) {
            // Map choice to label
            let choiceLabel = choice;
            if (choice === 'Emily Lewinsky') {
                choiceLabel = 'Selected Emily Lewinsky (Best Match)';
            } else if (choice === 'other broker') {
                choiceLabel = 'Requested to contact another broker';
            } else if (choice === 'not now') {
                choiceLabel = 'Not now — thinking it over';
            } else {
                choiceLabel = 'Selected: ' + choice;
            }

            // Prepare full payload
            const payload = {
                _subject: '📋 Full Client Feedback & Broker Selection',
                full_name: storedFeedback.fullName,
                trial_experience: storedFeedback.trialExperience,
                satisfaction: storedFeedback.satisfaction,
                willingness: storedFeedback.willingness,
                trading_goal: storedFeedback.tradingGoal,
                valued_traits: storedFeedback.valuedTraits.join(', '),
                broker_selection: choiceLabel,
                selection_timestamp: new Date().toISOString()
            };

            // Disable buttons to prevent double click
            const btns = document.querySelectorAll('#brokerSection .btn-success, #brokerSection .btn-primary, #brokerSection .btn-outline');
            btns.forEach(btn => btn.disabled = true);

            // Send to Formspree
            const endpoint = 'https://formspree.io/f/mrenbayp';
            fetch(endpoint, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(payload)
            })
            .then(response => {
                if (!response.ok) {
                    throw new Error('Server responded with status ' + response.status);
                }
                return response.json();
            })
            .then(data => {
                console.log('All data sent successfully:', data);
                // Show confirmation
                showConfirmation(choice);
            })
            .catch(error => {
                console.error('Error sending data:', error);
                alert('There was an error sending your data. Please try again later.\n\nError: ' + error.message);
                // Even if error, we still show confirmation to not block user
                showConfirmation(choice);
            })
            .finally(() => {
                btns.forEach(btn => btn.disabled = false);
            });
        }

        /**
         * Display the confirmation page with a tailored message
         */
        function showConfirmation(choice) {
            // Hide broker section, show confirmation
            document.getElementById('brokerSection').style.display = 'none';
            const confirmDiv = document.getElementById('confirmation');
            confirmDiv.style.display = 'block';
            updateSteps(3);
            confirmDiv.scrollIntoView({ behavior: 'smooth', block: 'center' });

            const title = document.getElementById('confirmationTitle');
            const msg = document.getElementById('confirmationMessage');

            if (choice === 'Emily Lewinsky') {
                title.textContent = '🎉 You Selected Emily Lewinsky!';
                msg.innerHTML = `Thank you, <strong>${storedFeedback.fullName}</strong>!<br>
                Emily is our top recommendation, and we're confident she'll help you achieve your goals.<br>
                Our team will contact you within 24 hours to arrange a dedicated onboarding session.`;
            } else if (choice === 'other broker') {
                title.textContent = '📞 We’ll Connect You with Another Broker';
                msg.innerHTML = `Thank you, <strong>${storedFeedback.fullName}</strong>!<br>
                Our advisors will reach out shortly to help you connect with the broker that best fits your needs.<br>
                You can expect to hear from us within 24 hours.`;
            } else if (choice === 'not now') {
                title.textContent = '⏳ Take Your Time';
                msg.innerHTML = `Thank you, <strong>${storedFeedback.fullName}</strong>!<br>
                We respect your decision. You can return to this page anytime to make a final choice.<br>
                Your client manager is also available to answer any questions.`;
            } else {
                title.textContent = 'Thank You!';
                msg.innerHTML = `Your feedback has been recorded. We'll get back to you shortly.`;
            }
        }

        // Smooth scroll for "Give Feedback" button
        document.querySelector('.notice-banner .btn')?.addEventListener('click', function(e) {
            e.preventDefault();
            document.getElementById('feedback').scrollIntoView({ behavior: 'smooth', block: 'start' });
        });

        // On load, ensure step 1 is active
        updateSteps(1);
    </script>
